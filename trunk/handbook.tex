\section[初等代数]{初~等~代~数}
\subsection{乘法及因式分解公式}
\begin{compactenum}[\indent(1)]
	\item $(x+a)(x+b)=x^2+(a+b)x+ab$.
	\item $(a \pm b)^2 = a^2 \pm 2ab + b^2$.
	\item $(a \pm b)^3 = a^3 \pm 3a^2b + 3ab^2 \pm b^3$.
	\item $(a+b+c)^2 = a^2 + b^2 + c^2 + 2ab + 2bc + 2ca$.
	\item $a^2 - b^2 = (a-b)(a+b)$.
	\item $a^3 \mp b^3 = (a \mp b)(a^2 \pm ab + b^2)$.
	\item $a^n - b^n = (a-b) (a^{n-1}+a^{n-2}b + a^{n-3}b^2 + \cdots + ab^{n-2} + b^{n-1})$.
	\item $a^n - b^n = (a+b) ( a^{n-1} - a^{n-2}b + a^{n-2}b^3 - \cdots + ab^{n-2} - b^{n-1}$，$n=$ 偶数。
	\item $a^n + b^n = (a+b) (a^{n-1} - a^{n-2}b + a^{n-3}b^2 - \cdots - ab^{n-2} + b^{n-1}$，$n=$ 奇数。
	\item $a^3 + b^3 + c^3 - 3ab= (a+b+c)(a^2 + b^2 + c^2 - ab - bc - ca)$.
\end{compactenum}

\subsubsection{分式运算}
\break
\begin{compactenum}[\indent 1)]
	\item \setts 加减法：\hspace*{6em}&$\nfrac{a}{b}\pm \nfrac{c}{b} = \nfrac{a \pm c}{b}$\endl
		\usets & $\nfrac{a}{b}\pm \nfrac{c}{d} = \nfrac{ad \pm bc}{bd}$\ok
	\item \usets 乘法：&$\nfrac{a}{b} \times \nfrac{c}{d} = \nfrac{a \times c}{b \times d}.$\ok
	\item \usets 除法：& $\nfrac{a}{b} \div \nfrac{c}{d} = \nfrac{a}{b} \times \nfrac{d}{c}.$\ok
	\item \usets 乘方：& $\left( \dfrac{a}{b}\right)^n = \dfrac{a^n }{b^n}.$\ok
	\item \usets 开方：& $\sqrt[n]{\dfrac{a}{b}} = \nfrac{\sqrt[n]{a}}{\sqrt[n]{b}}.$\ok
\end{compactenum}
\subsubsection{分项分式}
在以下各式中，等号左边都是真分式：
\begin{compactenum}[\indent 1)]
	\item $\nfrac{A}{(x-a)(x-b)(x-c)} = \nfrac{A_1}{x-a} + \nfrac{A_2}{x-b} + \nfrac{A_3}{x-c},$\\
	\centerline{$A_1, A_2, A_3$都是常数。}
	\item $\nfrac{A}{(x-a)^3} = \nfrac{A_1}{x-a} + \nfrac{A_2}{(x-a)^2} + \nfrac{A_3}{(x-a)^3},$\\
	\centerline{$A_1, A_2, A_3$ 都是常数。}
	\item $\nfrac{A}{PQR^3} = \nfrac{A_1}{P} + \nfrac{A_2}{Q} + \nfrac{B_1}{R} + \nfrac{B_2}{R^2} + 
		\nfrac{B_3}{R^3},$\\[2pt]
	\setts 其中\quad & $P,Q,R$ 是一次式或二次质因式，\endl
	\usets &$A_1, A_2$ 分别比\ $P, Q$ 的次数低一次，\endl
	\usets &$B_1, B_2, B_3$ 都比\ $R$ 的次数低一次。\ok
\end{compactenum}
\subsection[比例]{比~~例}
设\ $a:b = c:d$ 或\ $\nfrac{a}{b} = \nfrac{c}{d}$， 则：
\begin{compactenum}[\indent (1)]
	\item $ad = bc$ （内项积等于外项积）。
	\item $b:a = d:c$ （反比也相等）。
	\item $a:c = b:d$ \quad $d:b = c:a$ （更比）。
	\item \hspace*{5em}$\nfrac{a+b}{b} = \nfrac{c+d}{d}$ （合比）。
	\item \hspace*{5em}$\nfrac{a-b}{b} = \nfrac{c-d}{d}$ （分比）。
	\item \hspace*{5em}$\nfrac{a+b}{a-b} = \nfrac{c+d}{c-d}$ （合分比）。
	\item 设 \hspace*{3em} $\nfrac{a}{b} = \nfrac{c}{d} = \nfrac{e}{f} $，则：
	$$\frac{a}{b} = \frac{la + mc + ne}{lb + md + nf} = \frac{\sqrt{a^2 + c^2 + e^2}}{\sqrt{b^2 + d^2 + f^2}}.$$
	\item 若\ $y$ 与\ $x$ 成正比（或写为\ $y \propto x$），则：\\
	$\nfrac{y}{x} = k$ 或\ $y = kx$ （其中\ $k$ 为比例常数，以下同）。
	\item 若\ $y$ 与\ $x$ 成反比$\Big($或写为\ $y \propto \nfrac{1}{x}\Big)$，则：
		$$y:\frac{1}{x} = k \text{~或~} xy=k$$
	\item 若\ $y$ 与\ $x$ 成正比，$y$ 与\ $z$ 也成正比（或写为\ $y \propto x,\, y\propto z$），则：
		$$y \propto xz \text{~或~} y=kxz$$
\end{compactenum}
\subsection{不等式}
\subsubsection{基本不等式}
设\ $a>b$，（1--4），则：
\begin{compactenum}[\indent 1)]
	\item \setts \hspace*{6em} & $ac>bc$, \quad & $(c>0)$.\endl
		\usets & $ac<bc$, & $(c<0)$.\ok
	\item \usets & $\dfrac{a}{c} > \dfrac{b}{c}$, & $(c>0)$.\endl
		\usets & $\dfrac{a}{c} < \dfrac{b}{c}$, & $(c<0)$.\ok
	\item \hfill $a^n>b^n$,\quad $(n>0,\,a>0,\,b>0)$. \hfill~
	\item \hfill $\sqrt[n]{a}>\sqrt[n]{b}$,\quad $(n\,\text{为正整数},\,a>b>0)$. \hfill~
	\item 设\ $\nfrac{a}{b}<\nfrac{c}{d}$，且\ $b$，$d$ 同号则：\\
		\usets & $\nfrac{a}{b}<\nfrac{a+c}{b+d}<\nfrac{c}{d}$.\ok
\end{compactenum}
\subsubsection{绝对不等式}
设以下各量都是正值，则：
\begin{compactenum}[\indent 1)]
	\item \setts \hspace*{3em} & $\nfrac{a+b}{2} \geqslant \sqrt{ab}$, $\nfrac{a+b+c}{3}\geqslant\sqrt[3]{abc}$，\endl
		\usets & $\nfrac{a+b+c+\dots+l}{n}\geqslant\sqrt[n]{abc\times\dots\times l}$，\endl
		\usets & \quad $(a,b,c,\dots,l\text{ 共\ }n\text{ 个，下同})$。\ok
	\item $\left(\dfrac{a\!+\!b\!+\!c\!+\!\dots\!+\!l}{n}\right)^r\! \lessgtr
		\dfrac{a^r\!+\!b^r\!+\!c^r\!+\!\dots\!+\!l^r}{n}$，
		$(r\!\lessgtr\! l)$。
\end{compactenum}
\subsubsection{绝对值与不等式}
\begin{compactenum}[\indent ~1)]
	\item $\lvert A+B\rvert \leqslant \lvert A\rvert + \lvert B\rvert$.
	\item $|A-B|\leqslant |A|+|B|$.
	\item $|A-B|\geqslant |A|-|B|$.
	\item $-|A|\leqslant A \leqslant |A|$.
	\item $\sqrt{A^2}=|A|$，（根号规约）。
	\item $|AB|=|A|\,|B|$，$\left\lvert \dfrac{A}{B}\right\rvert = \nfrac{|A|}{|B|}$.
	\item 若\ $|A|\leqslant B$，而\ $B>0$，则：$-B\leqslant A \leqslant B$.
\end{compactenum}
